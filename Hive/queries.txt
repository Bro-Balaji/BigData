SELECT PK.PKEY_NAME 4_PARTITION_NAME,P.PART_NAME 3_PARTITION_FULL_NAME,
 PKV.PART_KEY_VAL 5_DATE_VALUE,T.TBL_NAME 2_TBL_NAME,D.NAME 1_DB_NAME
  FROM PARTITION_KEY_VALS PKV
  JOIN
   PARTITIONS P ON P.PART_ID = PKV.PART_ID
   JOIN
    TBLS T ON T.TBL_ID = P.TBL_ID
     JOIN
      PARTITION_KEYS PK ON PK.TBL_ID = T.TBL_ID
      JOIN
       DBS D ON D.DB_ID = T.DB_ID
        WHERE PKV.PART_KEY_VAL < '2023-06-15'
        ORDER BY D.NAME,T.TBL_NAME
